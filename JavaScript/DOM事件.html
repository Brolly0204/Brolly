# JS中事件对象

    // 在每个监听函数里面 浏览器会默认给它传递一个对象数据类型的值 称为事件对象
    // 事件对象用来存储关于 当前事件 信息
    /**
     * 事件对象 存储事件的信息 一个对象数据类型的值
     * e || window.event
     * e.type 事件类型
     * e.target || e.srcElement事件源
     * e.clientX 距离窗口的X轴坐标
     * e.clientY 距离窗口的Y轴坐标
     * e.pageX = e.pageX || (e.clientX + sLeft); 距离body的X轴坐标 
     * e.pageY = e.pageY || (e.clientY + sTop) 距离body的Y轴坐标
     * e.offsetX 距离当前元素的X轴坐标
     * e.offsetY 距离当前元素的Y轴坐标
     * e.stopPropagation() || e.cancelBubble = true 阻止冒泡传播
     * e.preventDefault() || e.returnValue = false 阻止默认行为
     * e.keyCode/e.which 键码值
     * e.key 键码符
     * e.wheelDelta 滚轮方向
     */

    // 从当前元素开始 往外冒泡

      console.log('=> inner click', event.target)
    }

    }

      console.log('=> => => body click', event.target)
    }

      console.log('=> => => => html click', event.target)
    }
    document.onclick = function(event) {
      console.log('=> => => => => document click', event.target)
    }
    window.onclick = function(event) {
      console.log('=> => => => => => window click', event.target)
    }
  </script>
</body>
</html>

    var oUl = document.getElementById('oUl');

# 鼠标跟随

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Document</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }

    .box {
      position: relative;
      width: 400px;
      height: 400px;
      border: 1px solid red;
      margin: 50px 100px;
    }

    .mask {
      display: none;
      position: absolute;
      left: 0;
      top: 0;
      /* width: 100px;
      height: 100px; */
      background: rgba(0, 0, 0, .3);
      cursor: move;
    }
  </style>
</head>
<body>
  <div id="box" class="box">
    <div id="mask" style="width: 100px; height: 100px;" class="mask"></div>
  </div>
  <script>
    const $ = selector => document.getElementById(selector)

    const box = $('box')
    const mask = $('mask')
   
    const { height, width } = mask.style
    const { clientWidth, clientHeight, offsetLeft, offsetTop } = box


    const maxL = clientWidth - parseFloat(width)
    const maxT = clientHeight - parseFloat(height)

    box.onmouseenter = function() {
      mask.style.display = 'block'
    }
    
    box.onmousemove = function (e) {
        let l = e.clientX - offsetLeft - parseFloat(width) / 2
        let t = e.clientY - offsetTop - parseFloat(height) / 2

        if (l < 0) {
          l = 0
        } else if (l > maxL) {
          console.log('超出左边 最大边界')
          l = maxL
        }

        if (t < 0) {
          t = 0
        } else if (t > maxT) {
          t = maxT
        }


        mask.style.left = l + 'px'
        mask.style.top = t + 'px'
    }

    box.onmouseleave = function(e) {
      mask.style.display = 'none'
    }
  </script>
</body>
</html>

  <script>
    const $ = selector => document.querySelector(selector)
    const win = attr => document.documentElement[attr] || document.body[attr]
    
    const box = $('#box')
    box.onmousedown = function(e) {
      this.offsetX = e.offsetX
      this.offsetY = e.offsetY
      // this.offsetX = e.clientX - this.offsetLeft

      this.maxL = win('clientWidth') - this.offsetWidth
      this.maxT = win('clientHeight') - this.offsetHeight

      document.onmousemove = move.bind(this)
      document.onmouseup = up
    }

    function move(e) {
      let l = e.clientX - this.offsetX
      let t = e.clientY - this.offsetY

      l < 0 && (l = 0)
      l > this.maxL && (l = this.maxL)

      t < 0 && (t = 0)
      t > this.maxT && (t = this.maxT)
