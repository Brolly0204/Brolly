# 类名操作

# classList

classList是元素对象自带的内置属性，该属性提供了用于在元素中添加，移除及切换CSS类名的方法。

// 是否包含类名
ele.classList.contains(class)

// 添加类名
ele.classList.add(class1, class2, ...)

// 移除类名
ele.classList.remove(class1, class2, ...)

// 切换类名  无就添加 有就移除
classList.toggle(class)

参考：https://developer.mozilla.org/zh-CN/docs/Web/API/Element/classList

# 将类名操作方法挂载到Element类原型上

    const box = document.getElementById('box')

    // 方便元素使用挂载到 Element类的原型上
    // 所有标签元素都有一个共同类Element

    Element.prototype.hasClass =  function (cn){
      return this.className.includes(cn)
    }

    Element.prototype.addClass = function (...rest) {
      rest.forEach(c => {
        if (!this.hasClass(c)) {
          this.className += ` ${c}`
        }
      })
      return this
    }

    Element.prototype.removeClass = function (...rest) {
      rest.forEach(c => {
        let reg = new RegExp(`\\b${c}\\b`)
        this.className = this.className.replace(reg, '')
      })
      return this
    }

    Element.prototype.toggleClass =  function (...rest) {
      rest.forEach(cn => {
        this.hasClass(cn) ? this.removeClass(cn) : this.addClass(cn)
      })
      return this
    }

### 类名操作方法实现

> 类名的增删改查

### 1.hasClass是否包含指定类名

```js
  // hasClass 是否包含指定类名
  function hasClass(curEle, cn) {
    return new RegExp('\\b' + cn + '\\b').test(curEle.className)
  }
  hasClass(oDiv, 'li3')

# 2.addClass增加类名

function addClass(curEle, cn) {
    var list = cn.trim().split(/ +/); // 将指定类名拆分成数组
    for (var i = 0; i < list.length; i++) { // 一个一个进行检测添加
        cn = list[i];
        // !hasClass(curEle, cn) ? curEle.className += ' ' + cn : null;
        if(!hasClass(curEle, cn)){
            var str = curEle.className.trim();
            str += ' ' + cn;
            curEle.className = str.replace(/ +/g, ' ');
        }
    }
}

addClass(oDiv, 'li2 li3');

# 3.removeClass 移除类名

  function removeClass(curEle, cn) {
      var list = cn.trim().split(/ +/);
      for (var i = 0; i < list.length; i++) {
          cn = list[i];
          if (!hasClass(curEle, cn)) continue; // 检测下当前要移除的类名是否存在
          var reg = new RegExp('(^| +)' + cn + '( +|$)');
//         var reg = new RegExp('\\b'+cn+'\\b');
          var str = curEle.className.replace(reg, ' ').trim(); // 移除指定类名并且去除首尾空格
          curEle.className = str.replace(/ +/g, ' '); // 去除多余空格 保证类名格式规范
      }
  }

  removeClass(oDiv, 'li2 li3');

# 4.toggle 切换类名

  // toggle 切换类名
  function toggle(curEle, cn) {
      var list = cn.trim().split(/ +/);
      for (var i = 0; i < list.length; i++) {
          cn = list[i];
          hasClass(curEle, cn) ? removeClass(curEle, cn) : addClass(curEle, cn);
      }
  }
  toggle(oDiv, 'li3');

# 挂载到元素HTMLElement类的原型上

 /**
     * hasClass 检测是否存在指定类名
     * @param curEle 当前检测元素
     * @param cn 指定类名
     * @returns {boolean} true false
     */
    Element.prototype.hasClass = function (cn) {
        return new RegExp('\\b' + cn + '\\b').test(this.className);
    };

    //    console.log(oDiv.hasClass('li3'));


    /**
     * addClass 增加类名
     * @param curEle 当前元素
     * @param cn 指定类名
     */
    Element.prototype.addClass = function (cn) {
        var list = cn.myTrim().split(/ +/);
        for (var i = 0; i < list.length; i++) {
            cn = list[i];
            if (!this.hasClass(cn)) {
                var str = this.className.trim();
                str += ' ' + cn;
                this.className = str.replace(/ +/, ' ');
            }
        }
        return this;
    };

    //    console.log(oDiv.addClass('li3').hasClass('li3'));

    /**
     * removeClass 移除类名
     * @param curEle 当前元素
     * @param cn 指定类名
     */
    Element.prototype.removeClass = function (cn) {
        var list = cn.myTrim().split(/ +/);
        for (var i = 0; i < list.length; i++) {
            cn = list[i];
            if (this.hasClass(cn)) {
                var reg = new RegExp('(^| +)' + cn + '( +|$)');
                var str = this.className.replace(reg, ' ').trim();
                this.className = str.replace(/ +/, ' ');
            }
        }
        return this;
    };

    //    oDiv.removeClass('li1 li2');

    /**
     * toggle 切换类名
     * @param curEle 当前元素
     * @param cn 指定类名
     */
    Element.prototype.toggle = function (cn) {
        var list = cn.trim().split(/ +/);
        for (var i = 0; i < list.length; i++) {
            cn = list[i];
            this.hasClass(cn) ? this.removeClass(cn) : this.addClass(cn);
        }
        return this;
    };

    oDiv.toggle('li1 li3').addClass('li4').removeClass('li2')
