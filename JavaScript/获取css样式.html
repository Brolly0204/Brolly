# 获取css样式

# 一、获取行内样式

ele.style只能获取或设置行内样式

// 获取行内样式
ele.style.color

// 设置行内样式
ele.style.color = 'red'

# 二、获取最终渲染到页面中样式

获取最终经过浏览器计算过的样式（最终渲染到页面的样式）

# 1.window.getComputedStyle()

window.getComputedStyle() IE9以下的低版本 不支持这个属性方法

语法：


// let style = window.getComputedStyle(element, ['伪类']);

// 伪类一般用不到就设为null
let style = window.getComputedStyle(element, null);

// 获取宽度
style['width']
window.getComputedStyle(element, null)['width']

// 返回值style 是CSSStyleDeclaration的一个实例对象  
style instanceof CSSStyleDeclaration // true

CSSStyleDeclaration截图:

window.getComputedStyle(curEle, 伪类(不用要写成null)) 返回值 是CSSStyleDeclaration的一个实例对象 里面包含着 当前元素自身所有经过浏览器计算过的样式属性

获取伪类样式：

  .box::after {
    content: '文利';
  }

window.getComputedStyle(box, 'after')['content'] // '文利'

# 2.currentStyle

Element.currentStyle 是一个与 window.getComputedStyle方法功能相同的属性。这个属性只在在旧版本的IE浏览器中支持.

语法：

ele.currentStyle[attr]

# 获取样式兼容方法

  function getCss(ele, attr) {
    if ('getComputedStyle' in window) {
      return window.getComputedStyle(ele, null)[attr]
    } else {
      return ele.currentStyle[attr]
    }
  }
  getCss(ele, 'width')

1.针对IE低版本中透明度处理（filter滤镜）

.box {
  opacity: .1;
  filter: alpha(opacity=10); /* IE 9 以下*/
}

IE下获取透明度处理

// IE低版本 不支持window.getComputedStyle()，也不支持opacity样式属性
ele.currentStyle.filter // alpha(opacity=30)
ele.currentStyle['filter'] // alpha(opacity=30)

css filter（滤镜）属性 http://www.runoob.com/cssref/css3-pr-filter.html

  function getCss(ele, attr) {
    let value = null
    if ('getComputedStyle' in window) { // 标准浏览器
      value = window.getComputedStyle(ele, null)[attr]
    } else { // IE9以下低版本
      // 透明度 filter: alpha(opacity=30)
      if (attr === 'opacity') {
        // 如果是获取透明度 应该通过filter样式属性下获取
        const op = ele.currentStyle['filter']
        const reg = /^alpha\(opacity=(.*)\)$/
        value = reg.test(op) ? reg.exec(op)[1] / 100 : 1
      } else {
        value = ele.currentStyle[attr]
      }
    }
    return value
  }

2.将带有单位的样式值去单位

  function getCss(ele, attr) {
    let value = ''
    if ('getComputedStyle' in window) {
      // 标准浏览器
      value = window.getComputedStyle(ele, null)[attr]
    } else {
      // IE低版本浏览器
      if (attr === 'opacity') {
        let op = ele.currentStyle.filter // 'alpha(opacity=30)'
        let reg = /^alpha\(opacity=\)$/
        value = reg.test(op) ? reg.exec(op)[1] / 100 : 1 
      } else {
        value = ele.currentStyle[attr]
      }
    }
    // 把带单位（px pt em rem）的常见样式值 去单位处理
    let reg = /[+-]?(\d|[1-9]\d+)(\.\d+)?(px|pt|em|rem)$/
    // if (reg.test(value)) {
    //   value = parseFloat(value)
    // }
    return reg.test(value) ? parseFloat(value) : value
  }

  console.log(getCss(box, 'width')) // 100

# 设置样式

  • ele.style 是可读可写的（可获取、可设置）

  • window.getComputedStyle() 只能获取样式

  • ele.currentStyle 只读的 只能获取样式

设置css样式方法

  const box = document.getElementById('box')

  function setCss(ele, attr, value) {
    return ele.style[attr] = value
  }
  setCss(box, 'width', '300px')
  setCss(box, 'height', '300px')

1.针对IE低版本下透明度设置问题

IE低版本透明度需要通过filter样式属性设置设置

.box {
  opacity: .1;
  filter: alpha(opacity=10); /* IE 9 以下*/
}

  function setCss(ele, attr, value) {
    if (attr === 'opacity') {
      ele.style['opacity'] = value
      ele.style.filter = `alpha(opacity=${value * 100})`
      return
    }

    return ele.style[attr] = value
  }

2.针对其他版本float设置

  ele.style.cssFloat = 'right' // 兼容 低版本 ff chrome
  ele.style.styleFloat = 'right' // IE low

  function setCss(ele, attr, value) {
    if (attr === 'opacity') {
      ele.style['opacity'] = value
      ele.style.filter = `alpha(opacity=${value * 100})`
      return
    }

    if (attr === 'float') {
      ele.style.cssFloat = value
      ele.style.styleFloat = value
      return
    }
    
    return ele.style[attr] = value
  }

3.为常见样式自动添加单位

  function setCss(ele, attr, value) {
    if (attr === 'opacity') {
      ele.style['opacity'] = value
      ele.style.filter = `alpha(opacity=${value * 100})`
      return
    }

    if (attr === 'float') {
      ele.style.cssFloat = value
      ele.style.styleFloat = value
      return
    }

      // 将常用样式属性 如果传进来的值 如果是数字 帮它加上单位
      // attr width height marginLeft paddingTop left top bottom 等
      let reg = /^(height|width|((padding|margin)?(top|right|bottom|left)?))$/i
      
      // 设置的是不是常用带单位的样式属性
      if (reg.test(attr)) {
        // 如果是有效数字 就需要加单位
        if (!isNaN(value)) {
          value += 'px'
        }
      }
      ele.style[attr] = value
  }

# 批量设置样式

  // 批量设置css
  function setCssBatch(ele, options) {
    let isObj = Object.prototype.toString.call(options) === '[object Object]'
    if (isObj) {
      for (let attr in options) {
        if (options.hasOwnProperty(attr)) {
          setCss(ele, attr, options[attr])
        }
      }
    }
  }

# 设置样式、获取样式统一封装

  function css() {
    let len = arguments.length;
    let fn = getCss
    let isObj = Object.prototype.toString.call(arguments[1]) === '[object Object]'
    if (len >=3) {
      fn = setCss
    } else if(len === 2 && isObj) {
      fn = setCssBatch
    }
    return fn(arguments)
  }

    var oDiv = document.getElementById('oDiv');

    var op, l, t;
   oDiv.timer = setInterval(function () {
       op = utils.getCss(oDiv, 'opacity');
        l = utils.getCss(oDiv, 'left');
        t = utils.getCss(oDiv, 'top');
        if(op >=1){
            utils.setCss(oDiv, 'opacity', 1);

            return;

        t += 50;

           left: l,
           top: t
       })
    }, 300);
